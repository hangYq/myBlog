    # css实现左侧宽度固定，右侧自适应布局的四种方法

    # 1. 左侧浮动，右侧生成BFC

    	<div className = { style['float-left-center'] }>
        <div className={ style.left }></div>
        <div className={ style.right }>
            <p>左侧浮动的两栏内容</p>
            <p>左侧浮动的两栏内容</p>
            <p>左侧浮动的两栏内容</p>
            <p>左侧浮动的两栏内容</p>
        </div>
    </div>

    	// 左侧浮动，右侧生成BFC
	.float-left-center {
	    margin: 5px 0;
	    width: 100%;
	    color: blue;
	    .left {
	        float: left;
	        width: 300px;;
	        height: 100px;
	        background: red;
	    }
	    .right {
	        background: #eee;
	        //height: 100px;
	        overflow: auto;
	    }
	}

    # 2. flex布局实现

        <div className={ style['flex-left-center']}>
        <div className={ style.left }></div>
        <div className={ style.right }>
            <p>flex布局，左侧固定</p>
            <p>flex布局，左侧固定</p>
            <p>flex布局，左侧固定</p>
            <p>flex布局，左侧固定</p>
            <p>flex布局，左侧固定</p>
        </div>
    </div>

    	/*flex布局实现*/
	.flex-left-center {
	    display: flex;
	    margin: 5px 0;
	    .left {
	        width: 300px;
	        height: 100px;
	        background: red;
	    }
	    .right{
	        flex: 1;
	        background: #eee;
	    }
	}

    # 3. css3 网格布局

        <div className={ style['grid-left-center']}>
        <div className={ style.left }></div>
        <div className={ style.right }>
            <p>网格布局实现</p>
            <p>网格布局实现</p>
            <p>网格布局实现</p>
            <p>网格布局实现</p>
            <p>网格布局实现</p>
        </div>
    </div>

    	// 网格布局
	.grid-left-center {
	    margin: 5px 0;
	    display: grid;
	    grid-template-columns: 300px auto;
	    grid-template-rows: 200px auto;
	    
	    .left {
	        background: red;
	    }
	    .right {
	        background: #eee;
	    }
	}

    # 4. 绝对定位实现

        <div className={ style['absoulte-left-center']}>
        <div className={ style.left }></div>
        <div className={ style.right }>
            <p>绝对定位实现</p>
            <p>绝对定位实现</p>
            <p>绝对定位实现</p>
            <p>绝对定位实现</p>
            <p>绝对定位实现</p>
        </div>
    </div>

    	// 绝对定位实现
	.absoulte-left-center {
	    position: relative;
	    margin: 5px 0;
	    .left {
	        position: absolute;
	        width: 300px;
	        height: 100px;
	        background: red;
	    }
	    .right {
	        position: absolute;
	        left: 300px;
	        right: 0;
	        background: #eee;
	    }
	}
